#include <iostream>
using namespace std;

int main()
{
    int n;
    cout << "Enter number of elements: ";
    cin >> n;

    int A[n];

    cout << "Enter elements: ";
    for (int i = 0; i < n; i++)
        cin >> A[i];

    for (int i = 0; i < n - 1; i++)
    {
        bool flag = false;
        int min = i;

        for (int j = i + 1; j < n; j++)
        {
            if (A[j] < A[min])
            {
                min = j;
                flag = true;
            }
        }

        
        if (min != i)
        {
            swap(A[i], A[min]);
        }
    }

    cout << "Sorted array: ";
    for (int i = 0; i < n; i++)
        cout << A[i] << " ";

    return 0;
}
